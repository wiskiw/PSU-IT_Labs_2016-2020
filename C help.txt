cstdio

	=== Строки ===
	char * gets(char *buffer); - чтение строки в buffer; Функция возвращает значение параметра buffer, 
		если строка была введена, или значение NULL (ноль), 
		если был достигнут конец потока ввода. Данная функция осуществляет ввод строки, 
		пока не встретится символ перевода строки. 
		
	int puts(const char *string); - вывод строки; Функция возвращает целочисленное значение: 
		положительное число или ноль в случае успеха, и значение EOF (-1) в случае ошибки.
		
	int sscanf(const char * restrict buffer, const char * restrict string,[address] ...); 
		- Функция форматированного ввода из строки 
		
		
	int sprintf(char * restrict buffer, const char * restrict format, [argument] ...); 
		- Функции форматированного вывода в строку
		
	int snprintf(char * restrict buffer, size_t maxsize,
				 const char * restrict format, [argument] ...);
