#include <stdio.h>
#include <math.h>
#include <Windows.h>
int main()
{
	int i, n = 100;
	double a = 0.0, b = 1.0, h, sum = 0.0;

	h = (b - a) / n;
	for (i = 0; i <= n - 1; i++)
	{
		sum += pow((a + h / 2.0*(double)(1.0 + 2.0*i)), 9.0);
	}
	double intg = sum*h, del1 = intg - 0.1;
	n = 200;
	sum = 0.0;
	h = (b - a) / n;
	for (i = 0; i <= n - 1; i++)
	{
		sum += pow((a + h / 2.0*(double)(1.0 + 2.0*i)), 9.0);
	}
	intg = sum*h;
	double del2 = intg - 0.1;
	printf("%.16lf\ndelta1 = %.16lf\ndelta2 = %.16lf", intg, del1, del2);
	printf("\nResultat: %lf\n\n",log2(fabs(del1/del2)));
	system("pause");
}
